170  // something
171  do {
172  // While there's no data, run the timeout counter
173  RECEIVE_ERROR re;
174  uint32_t u32_count = 0;

177  u32_count++;
178  doHeartbeat();

180 
181  // Step the machine

183  if (u32_count >= RECEIVE_TIMEOUT)
